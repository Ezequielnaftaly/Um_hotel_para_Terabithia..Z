def programa_quartos():
    global nome_usuario
    global nome_hotel

    print("\nPrograma - Quantos quartos são?")
    valor_diaria = float(input("Qual o valor padrão da diária? "))
    if valor_diaria < 0:
        print(f"Valor inválido, {nome_usuario}.")
        return

    dias_hospedagem = int(input("Quantas diárias serão necessárias? "))
    if dias_hospedagem < 0 or dias_hospedagem > 30:
        print(f"Valor inválido, {nome_usuario}.")
        return

    print(f"O valor de {dias_hospedagem} dias de hospedagem é de R${valor_diaria * dias_hospedagem}")
    nome_hospede = input("Qual o nome do hóspede? ")
    confirmacao = input(f"{nome_usuario}, você confirma a hospedagem para {nome_hospede} por {dias_hospedagem} dias? (S/N) ")
    if confirmacao.upper() == "S":
        print(f"{nome_usuario}, reserva efetuada para {nome_hospede}. O valor total é de R${valor_diaria * dias_hospedagem}.")
    else:
        print(f"{nome_usuario}, reserva não efetuada.")
